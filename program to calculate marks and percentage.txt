//Write a program to input marks of 5 subject and calculate total marks and percentage.
#include<iostream>
using namespace std;
int main()
{
    int marks1,marks2,marks3,marks4,marks5,total;
    float percentage;
    cout<<"Enter marks of Physics:";
    cin>>marks1;
    cout<<"Enter marks of chemistry:";
    cin>>marks2;
    cout<<"Enter marks of Maths:";
    cin>>marks3;
    cout<<"Enter marks of Biology:";
    cin>>marks4;
    cout<<"Enter marks of English:";
    cin>>marks5;
    total=marks1+marks2+marks3+marks4+marks5;
    cout<<"Total marks of all subjects is:"<<total<<"\n";
    percentage=((total/500.0)*100);
    cout<<"Obtained percentage is:"<<percentage<<"\n";
    return 0;
}
