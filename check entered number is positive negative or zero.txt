//Write a program to input a number and check whether it is positive,negative or zero.
#include<iostream>
using namespace std;
int main()
{
    int num;
    cout<<"Enter any number:";
    cin>>num;
    if(num>0)
    {
        cout<<"Entered number is positive";
    }else if(num<0){
        cout<<"Entered number is negative";
    }else{
        cout<<"Entered number is zero(0)";
    }
    return 0;
}
