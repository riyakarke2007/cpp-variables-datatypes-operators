//Write a program to input number and check whether it is multiple of both 3 and 7.
#include<iostream>
using namespace std;
int main()
{
    int num;
    cout<<"Enter a number:";
    cin>>num;
    if(num%3==0 && num%7==0){
        cout<<"Entered number is multiple of both 3 as well as 7";
    }else{
        cout<<"Entered number is not multiple of 3 and 7";
        return 0;
    }
}
