//write a program to input two number and swap them using a third variable.
#include<iostream>
using namespace std;
int main()
{
    int a,b,temp;
    cout<<"Enter first number:";
    cin>>a;
    cout<<"Enter second number:";
    cin>>b;
    //Swapping using third variable.
    temp=a;
    a=b;
    b=temp;
    cout<<"After swapping first number is:"<<a<<"\n";
    cout<<"After swapping second number is:"<<b<<"\n";
    return 0;
}
