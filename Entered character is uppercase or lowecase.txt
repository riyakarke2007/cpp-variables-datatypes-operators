//Write a program to input a character and check whether it is uppercase,lowercaseor digit.
#include<iostream>
using namespace std;
int main()
{
    char ch;
    cout<<"Enter any character(uppercase) or (lowercase):";
    cin>>ch;
    if(ch>='a' && ch<='z'){
        cout<<"Entered character is lowercase";
    }
    else if(ch>='A' && ch<='Z'){
        cout<<"Entered character is uppercase";
    }else{
        cout<<"Invalid symbol";
}
return 0;
}
