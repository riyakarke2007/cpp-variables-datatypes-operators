//Write a program to input a number and check whether it is divisible by 5 and 11.
#include<iostream>
using namespace std;
int main()
{
    int num;
    cout<<"Enter a number:";
    cin>>num;
    if(num%5==0 && num%11==0)
    {
        cout<<"Enterd number is divisible by 5 as well as 11";
        
    }else{
        cout<<"Entered number is not divisible by 5 and 11";
        
    }
    return 0;
}
